-- What is the profit for each of the top product categories in the year 2021, sorted by profit in descending order?
-- Use OrderDate and OrderLineProfit


SELECT
	{[Measures].[Profit]} ON COLUMNS,
	Order(
		TopCount(
			[Product.ProductCategory].[All Product.ProductCategorys].Children,
			[Measures].[Profit]
		),
		[Measures].[Profit],
		DESC
	) ON ROWS
FROM
	[bike_sales]
WHERE
	[OrderDate].[Year].&[2021]
